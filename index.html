// ============================================================ // DONIA LABS TECH — AI Proxy Worker (Claude primary + Gemini fallback) // ضع هذا الكود في Cloudflare Worker الحالي "donia-ai-proxy" // (أو في Worker جديد بنفس الاسم إذا لم يكن موجوداً) // // المفاتيح لا توضع هنا أبداً كنص. تُضاف من لوحة Cloudflare كـ Secrets: // Workers & Pages → donia-ai-proxy → Settings → Variables and Secrets // 1) ANTHROPIC_API_KEY = مفتاح Claude الحالي // 2) GEMINI_API_KEY = مفتاح Gemini الجديد (يبدأ بـ AIza...) // هذا بالضبط ما يحقق طلبك: "تدخله مرة واحدة ويختفي ويعمل بالخلفية" — بأمان حقيقي. // ============================================================ export default { async fetch(request, env) { // CORS: السماح فقط من نطاقك const ALLOWED_ORIGINS = [ 'https://donialabstech.online', 'https://www.donialabstech.online', 'http://localhost:3000', // للتطوير المحلي فقط ]; const origin = request.headers.get('Origin') || ''; const isAllowed = ALLOWED_ORIGINS.some(o => origin.startsWith(o)); const corsHeaders = { 'Access-Control-Allow-Origin': isAllowed ? origin : 'null', 'Access-Control-Allow-Methods': 'POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', }; if (request.method === 'OPTIONS') { return new Response(null, { headers: corsHeaders }); } const url = new URL(request.url); // كلا المسارين يستخدمان نفس منطق Claude→Gemini fallback // /api/chat ← يستخدمه المساعد الذكي (الرد على العملاء) // /api/claude-blog ← يستخدمه وكيل المدونة (كتابة المقالات) if (url.pathname === '/api/chat' || url.pathname === '/api/claude-blog') { try { const body = await request.json(); const result = await callWithFallback(body, env); return new Response(JSON.stringify(result), { headers: { ...corsHeaders, 'Content-Type': 'application/json' }, }); } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }, }); } } return new Response('Not found', { status: 404, headers: corsHeaders }); }, }; // يحاول Claude أولاً؛ إذا فشل (انقطاع، حد استخدام، خطأ مفتاح...) يتحول تلقائياً لـ Gemini. // الشكل المُرجَع متوافق دائماً مع شكل استجابة Anthropic الأصلي // { content: [{ type: "text", text: "..." }] } — لذلك لا حاجة لتعديل أي كود في الموقع. async function callWithFallback(body, env) { try { return await callClaude(body, env); } catch (claudeErr) { console.warn('[Worker] Claude failed, falling back to Gemini:', claudeErr.message); try { return await callGemini(body, env); } catch (geminiErr) { throw new Error( 'كلا المزودين فشل — Claude: ' + claudeErr.message + ' | Gemini: ' + geminiErr.message ); } } } async function callClaude(body, env) { if (!env.ANTHROPIC_API_KEY) throw new Error('ANTHROPIC_API_KEY غير مُعرَّف'); const resp = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': env.ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', }, body: JSON.stringify({ model: body.model || 'claude-sonnet-4-20250514', max_tokens: body.max_tokens || 800, system: body.system || '', messages: body.messages || [], }), }); if (!resp.ok) throw new Error('Claude error ' + resp.status); const data = await resp.json(); const text = (data.content || []).map((b) => b.text || '').join('\n'); return { content: [{ type: 'text', text }], provider: 'claude' }; } async function callGemini(body, env) { if (!env.GEMINI_API_KEY) throw new Error('GEMINI_API_KEY غير مُعرَّف'); const model = 'gemini-1.5-flash'; const geminiMessages = (body.messages || []).map((m) => ({ role: m.role === 'assistant' ? 'model' : 'user', parts: [{ text: m.content }], })); const payload = { contents: geminiMessages }; if (body.system) { payload.systemInstruction = { parts: [{ text: body.system }] }; } const resp = await fetch( `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${env.GEMINI_API_KEY}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload), } ); if (!resp.ok) throw new Error('Gemini error ' + resp.status); const data = await resp.json(); const text = data.candidates?.[0]?.content?.parts?.map((p) => p.text).join('\n') || ''; return { content: [{ type: 'text', text }], provider: 'gemini' }; }
